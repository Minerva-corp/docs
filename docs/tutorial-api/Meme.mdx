## Example usage

```typescript ts2esm2cjs|{3}|{3}
import fetch from 'node-fetch';

const fetchMeme = async () {
    const response = await fetch("https://api.minerva.my.id/v1/meme").then(res => res.json());

    console.log(response);
}

fetchMeme();
```
